核心数据模型

// 碎片记录
Fragment {
id, content, type (text/voice), source_app,
created_at, is_processed
}

// 日记条目
DiaryEntry {
id, date, title (≤30字), content (markdown),
word_count, is_paginated, page_index,
local_path, obsidian_path, created_at
}

// PERMA 评估
PermaScore {
diary_id, positive_emotion, engagement,
relationships, meaning, accomplishment,
ai_review, suggestions
}
